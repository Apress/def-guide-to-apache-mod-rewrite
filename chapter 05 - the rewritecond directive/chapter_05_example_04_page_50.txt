RewriteEngine on
RewriteCond %{TIME_HOUR}%{TIME_MIN} >0700
RewriteCond %{TIME_HOUR}%{TIME_MIN} <1900
RewriteRule ^/page\.html$ page.day.html
RewriteRule ^/page\.html$ page.night.html
